# 2. Count Products per Category:
  # Command:
  db.products.aggregate([
  {
    $group: {
      _id: "$category",
      count: { $sum: 1 }
    }
  }
])

# 3. Product Names and Prices, Sorted by Price (Descending):
  # Command:
  db.products.aggregate([
  {
    $project: {
      _id: 0,
      name: 1,
      price: 1
    }
  },
  {
    $sort: {
      price: -1
    }
  }
])
